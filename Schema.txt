        user_id INT PK
        first_name VARCHAR(100)
        last_name VARCHAR(100)
user_role_id INT
        fantasy_league_id INT
        fantasy_team_id INT
}

        role_id INT

}


Table Players  {
        player_id INT PK
        first_name VARCHAR(100)
        midlle_name VARCHAR(100)
        last_name VARCHAR(100)
        date_of_birth DATE

        role ENUM

        team_id INT PK

Table Matches  {
        match_id INT PK

        umpire_2 INT

Table Tournaments  {
        tournament_id INT PK

}

Table Match_Scorecards  {
        scorecard_id INT PK
        match_id INT FK
        team_id INT FK

Table Player_Scores  {
        player_score_id INT PK

Table Fantasy_Leagues  {
        league_id INT PK
        league_name VARCHAR(255)

}

Table Fantasy_Teams  {
        fantasy_team_id INT PK
